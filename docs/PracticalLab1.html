import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error, r2_score,mean_absolute_error

# 4. Define 3 linear regression
# - median house value vs. median income
# - median house value vs. population
# - median house value vs. number of households

ProcessOutput = [
                    {
                        "x-axis":"median_income",
                        "y-axis":"median_house_value",
                        "Intercept":0,
                        "Coefficient":0,
                        "R-squared":0,
                        "MeanSquaredError":0,
                        "MeanAbsoluteError":0
                    },
                    {
                        "x-axis":"population",
                        "y-axis":"median_house_value",
                        "Intercept":0,
                        "Coefficient":0,
                        "R-squared":0,
                        "MeanSquaredError":0,
                        "MeanAbsoluteError":0
                    },
                    {
                        "x-axis":"households",
                        "y-axis":"median_house_value",
                        "Intercept":0,
                        "Coefficient":0,
                        "R-squared":0,
                        "MeanSquaredError":0,
                        "MeanAbsoluteError":0
                    }
                ]
for i in ProcessOutput:
    X = dataset[[i['x-axis']]]
    Y = dataset[i['y-axis']]
    X_train, X_test, y_train, y_test = train_test_split(X, Y, test_size=0.2, random_state=47)

    # Initialize & fit the linear regression model
    model = LinearRegression()
    model.fit(X_train, y_train)

    # Predict on the test data
    y_pred = model.predict(X_test)

    # Set required parameters : intercept, slope, MSE, MAE
    i['Intercept'] =  model.intercept_              # Calculate Intercept
    i['Coefficient'] = model.coef_[0]               # Calculate Coefficient
    i['R-squared'] = r2_score(y_test, y_pred)       # Calculate R-squared
    i['MeanSquaredError'] = mean_squared_error(y_test, y_pred)      # Calculate MSE
    i['MeanAbsoluteError'] = mean_absolute_error(y_test, y_pred)    # Calculate MAE
    
    y_pred = model.predict(X_test) 

    # Scatter plot of actual data
    plt.scatter(X_test, y_test, color='skyblue', label='Actual data',s=5)

    # Plot the regression line
    plt.plot(X_test, y_pred, color='blue', linewidth=1, label='Regression line')

    # 6. Plot the resulting line on top of the scatter data for each of the three models (three separate graphs), 
    # add the line parameters (intercept and slope), MSE and MAE to the graph (you can use a text box). 
    x_max = X_test.max().values[0] 
    y_max = y_test.max()

    plt.text(x_max * 0.98, y_max * 0.95, f'Intercept: {i['Intercept']:.2f}', fontsize=10, color='black',horizontalalignment='right')
    plt.text(x_max * 0.98, y_max * 0.90, f'Slope: {i['Coefficient']:.2f}', fontsize=10, color='black',horizontalalignment='right')
    plt.text(x_max * 0.98, y_max * 0.85, f'MSE: {i['MeanSquaredError']:.2f}', fontsize=10, color='black',horizontalalignment='right')
    plt.text(x_max * 0.98, y_max * 0.80, f'MAE: {i['MeanAbsoluteError']:.2f}', fontsize=10, color='black',horizontalalignment='right')

    # Add labels and title
    plt.xlabel(i['x-axis'])
    plt.ylabel(i['y-axis'])
    plt.title(f'Linear Regression: { i['y-axis'] } vs. { i['x-axis']}')

    # Display the plot with legend
    plt.legend()
    plt.show()
